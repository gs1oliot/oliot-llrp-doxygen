397 // If there is same TagReport with same EPCID, then replace it.
398 // If there is no such report, then just add it.
399 int ELFIN::AIOperation::addTagReportUnique(TagReportSet* pSet,
400  LLRP::CTagReportData* pRpt) {
401  std::pair<TagReportSet::iterator, bool> result = pSet->insert(pRpt);
402  if (result.second == true) {
403  return 0;
404  }
405  else {
406  LLRP::CTagReportData *pOldRpt = *(result.first);
407  if (pOldRpt->countAccessCommandOpSpecResult() != 0) {
408  std::vector<LLRP::CAccessCommandOpSpecResult *> pStayingReportVector;
409  // LLRP Spec does not specifies that how many times to apply AccessSpec to each tags.
410  // So in this implementation, just apply AccessSpec every round in AIOperation loop,
411  // and if there exists same reports or same OpSpecID, then overwrite
412  int stayingReport = 1;
413  for (std::list<LLRP::CParameter *>::iterator old_iter = pOldRpt->beginAccessCommandOpSpecResult();
414  old_iter != pOldRpt->endAccessCommandOpSpecResult(); old_iter++) {
415  LLRP::CAccessCommandOpSpecResult *pOldACOpSpecResult = (LLRP::CAccessCommandOpSpecResult *) (*old_iter);
416  for (std::list<LLRP::CParameter *>::iterator new_iter = pRpt->beginAccessCommandOpSpecResult();
417  new_iter != pRpt->endAccessCommandOpSpecResult(); new_iter++) {
418  LLRP::CAccessCommandOpSpecResult *pNewACOpSpecResult = (LLRP::CAccessCommandOpSpecResult *) (*new_iter);
419  // If the AccessCommandOpSpecResults in old report are different from existing ones, then add them.
420  // But if OpSpecID is same, then ignore them.
421  int compareResult = AccessOperation::compareACOpSpecResult(pOldACOpSpecResult, pNewACOpSpecResult);
422  if (compareResult == 0) {
423  // Two ACOpSpecResults are same, so ignore.
424  stayingReport = 0;
425  } else if (compareResult != 0 &&
426  AccessOperation::getACOpSpecResultOPSpecID(pOldACOpSpecResult) ==
427  AccessOperation::getACOpSpecResultOPSpecID(pNewACOpSpecResult)) {
428  // Two ACOpSpecResults are different, but OpSpecID is same, so ignore.
429  stayingReport = 0;
430  }
431  }
432  if (stayingReport == 1) {
433  pOldRpt->removeSubParameterFromAllList((LLRP::CParameter *) pOldACOpSpecResult);
434  pStayingReportVector.push_back(pOldACOpSpecResult);
435  }
436  }
437  for (std::vector<LLRP::CAccessCommandOpSpecResult *>::iterator stay_iter = pStayingReportVector.begin();
438  stay_iter != pStayingReportVector.end();stay_iter++) {
439  pRpt->addAccessCommandOpSpecResult((LLRP::CParameter *) (*stay_iter));
440  }
441  }
442 
443  pSet->erase(result.first);
444  delete pOldRpt;
445  std::pair<TagReportSet::iterator, bool> result2 = pSet->insert(pRpt);
446  if (result2.second == true) {
447  return 1;
448  }
449  else {
450  // NOT_REACHED
451  return -1;
452  }
453  }
454 
455  return -1;
