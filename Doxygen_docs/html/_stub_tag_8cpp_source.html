106 void ELFIN::StubTag::setLogEPCStr() {
107  unsigned char hexchars[] = { '0', '1', '2', '3', '4',
108  '5', '6', '7', '8', '9', 'A', 'B', 'C', 'D', 'E', 'F' };
109  for (unsigned int i = 0; i < 12; i++) {
110  // Start from 2nd memory space. 0~1 spaces are for CRC bits,
111  // but cannot be obtained from PR9200 using auto_read2 API, so leave them as 0x00.
112  int ch = this->_MB[1]->at(i + 4);
113  _logEpcStr[i * 2] = hexchars[(ch & 0xf0) >> 4];
114  _logEpcStr[i * 2 + 1] = hexchars[ch & 0x0f];
115  }
116  _logEpcStr[24] = '\0';
117 }
